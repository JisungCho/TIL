package swing;

import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.File;

import javax.sound.sampled.AudioInputStream;

public class TicTacToeSingle2 {
	static class MyFrame extends JFrame implements ActionListener{
		
		static JPanel panelNorth ;
		static JPanel panelCenter;
		static JLabel labelTitle; //게임 문구 표현 , 승리했을때 문구
		static JButton[][] buttons = new JButton[3][3]; //버튼을 2차배열로

		static boolean buttonstop;
		static boolean end;

			super(title);
			this.setLayout(new BorderLayout());
			this.setSize(300, 400);
			this.setVisible(true);
			this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
			
			initUI(); // GUI초기화
			playSound("bgm.wav");
		}
		
		@Override
		public void actionPerformed(ActionEvent e) { // 버튼 클릭시 함수 콜 발생
			JButton btn = (JButton) e.getSource(); // 눌린 버튼 객체를 가져오는것 (Vs 버튼의 타이틀은 e.getActionCommand)
			
			int x = getButtonIndexX(btn);
			int y = getButtonIndexY(btn);
			if(map[x][y] != 0) {
				return;
			}
			playSound("button1.wav");
			//게임로직
			if(buttonstop==false) { // 버튼클릭을 멈출것인지 아닌지

					if(btn.getText().equals(" ")) {
						btn.setText("X");
						map[x][y] = 1;

						end = checkedX(map);
						player=0;

							playSound("Bingo.wav");
							labelTitle.setText("WIN X");
							panelNorth.add(labelTitle);
							this.add("North", panelNorth);
							buttonstop = true;
						}
						
					}
				}else { // O차례
					if(btn.getText().equals(" ")) {
						btn.setText("O");
						map[x][y] = 2;
						//System.out.println("map["+x+"]["+y+"] ="+map[x][y]);
						end = checkedO(map);
						player=1;
						if(end) {
							playSound("Bingo.wav");
							labelTitle.setText("WIN O");
							panelNorth.add(labelTitle);
							this.add("North", panelNorth);
							buttonstop = true;
						}
					}
				}
			
			}
			
			if(count == 9 && end==false) { // 가득찬 상태이면서 결과가 무승부
				labelTitle.setText("Draw");
				panelNorth.add(labelTitle);
				this.add("North", panelNorth);
				buttonstop = true;
			}
		}
		
		boolean checkedX(int[][] map) {
			if(map[0][0]==1 && map[0][1] ==1 && map[0][2] ==1) {
				return true;
			}else if(map[0][0]==1 && map[1][0] ==1 && map[2][0] ==1){
				return true;
			}else if(map[0][0]==1 && map[1][1] ==1 && map[2][2] ==1){

				return true;

				return true;
			}else if(map[0][1]==1 && map[1][1] ==1 && map[2][1] ==1){
				return true;

				return false;

		}
		boolean checkedO(int[][] map) {
			if(map[0][0]==2 && map[0][1] ==2 && map[0][2] ==2) {
				return true;
			}else if(map[0][0]==2 && map[1][0] ==2 && map[2][0] ==2){
				return true;
			}else if(map[0][0]==2 && map[1][1] ==2 && map[2][2] ==2){
				return true;
			}else if(map[0][2]==2 && map[1][2] ==2 && map[2][2] ==2){
				return true;
			}else if(map[0][2]==2 && map[1][1] ==2 && map[2][0] ==2){

			}else if(map[1][0]==2 && map[1][1] ==2 && map[1][2] ==2){
				return true;
			}else if(map[2][0]==2 && map[2][1] ==2 && map[2][2] ==2){
				return true;
			}else if(map[0][1]==2 && map[1][1] ==2 && map[2][1] ==2){

		int getButtonIndexX(JButton btn) {
			for(int i=0;i<buttons.length;i++) {
				for(int j=0;j<buttons[i].length;j++) {
					if(buttons[i][j] == btn) { //같은 주소값이면 같은 객체
						return i;
					}
				}
			}
			return 0; //못찾은 경우 0을 넘겨줌
		}
		
		int getButtonIndexY(JButton btn) {
			for(int i=0;i<buttons.length;i++) {
				for(int j=0;j<buttons[i].length;j++) {
					if(buttons[i][j] == btn) { //같은 주소값이면 같은 객체
						return j;
					}
				}
			}
			return 0; //못찾은 경우 0을 넘겨줌
		}
		
		public static void playSound(String fileName) {
			File file = new File("./wav/" + fileName);
			if(file.exists()) { // 파일이 존재하는가 T/F
				try {
					//InputStream : 앱의 (입력/출력)을 담당하는 클래스
					//해당 파일에 관한 InputStream 가져옴
					AudioInputStream stream = AudioSystem.getAudioInputStream(file);
					// Clip : 오디오파일 하나를 관리해줌.
					Clip clip = AudioSystem.getClip();
					clip.open(stream);
					if(fileName.equals("bgm.wav")) {
						clip.loop(Clip.LOOP_CONTINUOUSLY);
					}
					//clip.loop(Clip.LOOP_CONTINUOUSLY); //무한반복
					//clip.loop(2); //루프를 2번더 돌림
					clip.start();
				} catch (Exception e) {
					e.printStackTrace(); // 에러 경로 출력
				}
			}else {
				System.out.println("파일이 없습니다.");
			}
		}
		
		
		void initUI() {// GUI초기화
			panelNorth = new JPanel();
			panelNorth.setPreferredSize(new Dimension(300,100));
			panelNorth.setBackground(Color.DARK_GRAY);
			panelNorth.setLayout(new FlowLayout());
			
			labelTitle = new JLabel("Input O,X");
			labelTitle.setPreferredSize(new Dimension(300, 100));
			labelTitle.setFont(new Font("D2Coding", Font.BOLD, 20));//폰트설정
			labelTitle.setForeground(Color.white);//글자색
			labelTitle.setHorizontalAlignment(JLabel.CENTER); //글자 가운데 정렬
			
			panelNorth.add(labelTitle); //패널에 추가
			
			this.add("North", panelNorth); //프레임에 추가
			
			panelCenter = new JPanel();
			panelCenter.setPreferredSize(new Dimension(300,300));
			panelCenter.setLayout(new GridLayout(3,3));
			
			for(int i=0;i<buttons.length;i++) {
				for(int j=0;j<buttons[i].length;j++) {
					buttons[i][j] = new JButton(" ");
					buttons[i][j].setPreferredSize(new Dimension(100, 100));
					buttons[i][j].setFont(new Font("D2Coding", Font.BOLD, 50));
					buttons[i][j].setFocusable(false); // 포커스를 삭제해줌
					buttons[i][j].setBackground(Color.white);
					buttons[i][j].addActionListener(this); // 이 클래스에서 처리
					
					panelCenter.add(buttons[i][j]);//버튼을 패널에 하나씩 추가
				}
			}
			
			this.add("Center", panelCenter);
			
			this.pack();// 여백 공간을 없애줌.
			
			
		}
	}
	
	public static void main(String[] args) {
		new MyFrame("TicTacToe Game"); 
		
	}
}

package swing;

import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

import javax.swing.*;


public class TicTacToeSingle {
	static class MyFrame extends JFrame implements ActionListener{
		
		static JPanel panelNorth ;
		static JPanel panelCenter;
		static JLabel labelTitle; //게임 문구 표현 , 승리했을때 문구
		static JButton[][] buttons = new JButton[3][3]; //버튼을 2차배열로
		static int[][] map = new int[3][3];
		static int ClickCount = 0;
		static boolean turn = false; // false: X차례 , true : O차례
		static Timer timer = null;
		
		public MyFrame(String title) {
			super(title);
			this.setLayout(new BorderLayout());
			this.setSize(300, 400);
			this.setVisible(true);
			this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
			
			initUI(); // GUI초기화
		}
		
		@Override
		public void actionPerformed(ActionEvent e) { // 버튼 클릭시 함수 콜 발생
			JButton btn = (JButton) e.getSource(); // 눌린 버튼 객체를 가져오는것 (Vs 버튼의 타이틀은 e.getActionCommand)

			int x = getButtonIndexX(btn);
			int y = getButtonIndexY(btn);
			
			//중복 클릭 방지 
			if(map[x][y] != 0) {
				return;
			}
			
			//게임로직
			if(turn == false) { //X차례
				map[x][y] = 1;
				turn = true; //차례 바꿈
			}else { //O차례
				map[x][y] = 2;
				turn = false;//차례 바꿈
			}
			
			updateUI();
			
			ClickCount++; //눌린 카운트 추가
			
			//판정로직
			int result = judge();
			if(result == 1) {
				labelTitle.setText("Win X");
				doReset();
			}else if(result == 2) {
				labelTitle.setText("Win O");
				doReset();
			}
			else if( result == 0) {
				if(ClickCount == 9) {
					labelTitle.setText("Draw");
					doReset();
				}
			}
		}
		
		void doReset() {
			timer = new Timer(1000, new ActionListener() {
				@Override
				public void actionPerformed(ActionEvent e) {
					resetUI();
					timer.stop();
					timer=null;
				}
			});
			timer.start(); //타이머 시작
		}
		void resetUI() {//게임초기화
			labelTitle.setText("Start OX Game");
			for(int i=0;i<3;i++) {
				for(int j=0;j<3;j++) {
					map[i][j] = 0;
				}
			}
			turn = false;
			ClickCount = 0;
			
			updateUI();
			
		}
		
		int judge() { //누가 이겼는지 판단
			int result =0; // 0이면 진행중, 1이면 x승 ,2이면 0승
			//가로
			if( map[0][0] == 1 && map[0][1]==1 &&map[0][2]==1) { //모두 X인가
				result =1;
			}
			if( map[0][0] == 2 && map[0][1]==2 &&map[0][2]==2) { //모두 O인가
				result=2;
			}
			if( map[1][0] == 1 && map[1][1]==1 &&map[1][2]==1) { //모두 X인가
				result =1;
			}
			if( map[1][0] == 2 && map[1][1]==2 &&map[1][2]==2) { //모두 O인가
				result=2;
			}
			if( map[2][0] == 1 && map[2][1]==1 &&map[2][2]==1) { //모두 X인가
				result =1;
			}
			if( map[2][0] == 2 && map[2][1]==2 &&map[2][2]==2) { //모두 O인가
				result=2;
			}
			//세로
			if( map[0][0] == 1 && map[1][0]==1 &&map[2][0]==1) { //모두 X인가
				result =1;
			}
			if( map[0][0] == 2 && map[1][0]==2 &&map[2][0]==2) { //모두 O인가
				result=2;
			}
			if( map[0][1] == 1 && map[1][1]==1 &&map[2][1]==1) { //모두 X인가
				result =1;
			}
			if( map[0][1] == 2 && map[1][1]==2 &&map[2][1]==2) { //모두 O인가
				result=2;
			}
			if( map[0][2] == 1 && map[1][2]==1 &&map[2][2]==1) { //모두 X인가
				result =1;
			}
			if( map[0][2] == 2 && map[1][2]==2 &&map[2][2]==2) { //모두 O인가
				
			}
			//대각선
			if( map[0][0] == 1 && map[1][1]==1 &&map[2][2]==1) { //모두 X인가
				result =1;
			}
			if( map[0][0] == 2 && map[1][1]==2 &&map[2][2]==2) { //모두 O인가
				result=2;
			}
			//역대각선
			if( map[0][2] == 1 && map[1][1]==1 &&map[2][0]==1) { //모두 X인가
				result =1;
			}
			if( map[0][2] == 2 && map[1][1]==2 &&map[2][0]==2) { //모두 O인가
				result=2;
			}
			return result;
		}
		
		
		
		void updateUI() {
			for(int i=0;i<3;i++) {
				for(int j=0;j<3;j++) {
					if(map[i][j] == 0) {
						buttons[i][j].setText(" ");
					}else if(map[i][j]==1) {
						buttons[i][j].setText("X");
					}else if(map[i][j]==2) {
						buttons[i][j].setText("O");
					}
				}
			}
		}
		
		
		int getButtonIndexX(JButton btn) {
			for(int i=0;i<buttons.length;i++) {
				for(int j=0;j<buttons[i].length;j++) {
					if(buttons[i][j] == btn) { //같은 주소값이면 같은 객체
						return i;
					}
				}
			}
			return 0; //못찾은 경우 0을 넘겨줌
		}
		
		int getButtonIndexY(JButton btn) {
			for(int i=0;i<buttons.length;i++) {
				for(int j=0;j<buttons[i].length;j++) {
					if(buttons[i][j] == btn) { //같은 주소값이면 같은 객체
						return j;
					}
				}
			}
			return 0; //못찾은 경우 0을 넘겨줌
		}
		
		void initUI() {// GUI초기화
			panelNorth = new JPanel();
			panelNorth.setPreferredSize(new Dimension(300,100));
			panelNorth.setBackground(Color.DARK_GRAY);
			panelNorth.setLayout(new FlowLayout());
			
			labelTitle = new JLabel("Input O,X");
			labelTitle.setPreferredSize(new Dimension(300, 100));
			labelTitle.setFont(new Font("D2Coding", Font.BOLD, 20));//폰트설정
			labelTitle.setForeground(Color.white);//글자색
			labelTitle.setHorizontalAlignment(JLabel.CENTER); //글자 가운데 정렬
			
			panelNorth.add(labelTitle); //패널에 추가
			
			this.add("North", panelNorth); //프레임에 추가
			
			panelCenter = new JPanel();
			panelCenter.setPreferredSize(new Dimension(300,300));
			panelCenter.setLayout(new GridLayout(3,3));
			
			for(int i=0;i<buttons.length;i++) {
				for(int j=0;j<buttons[i].length;j++) {
					buttons[i][j] = new JButton(" ");
					buttons[i][j].setPreferredSize(new Dimension(100, 100));
					buttons[i][j].setFont(new Font("D2Coding", Font.BOLD, 50));
					buttons[i][j].setFocusable(false); // 포커스를 삭제해줌
					buttons[i][j].setBackground(Color.white);
					buttons[i][j].addActionListener(this); // 이 클래스에서 처리
					
					panelCenter.add(buttons[i][j]);//버튼을 패널에 하나씩 추가
				}
			}
			
			this.add("Center", panelCenter);
			
			this.pack();// 여백 공간을 없애줌.
			
			//버튼에 글자 넣기 
			
		}
	}
	
	public static void main(String[] args) {
		new MyFrame("TicTacToe Game"); 
	}

2. 선언부와 실행부

   ```java
   //public : 클래스 접근 권한(모두에게 오픈)
   //class : 클래스 예약어(KeyWord),변수이름으로 쓸 수 없음
   //ex2_02 : 클래스 이름(파일이름과 동일)
   // { ... } : 코드의 영역
   public class ex2_02 {
   	//선언부
   	
   	//public : 함수의 접근 권한(모두에게 오픈)
   	//static : 정적변수,정적함수를 의미함
   	//void : 함수의 반환값이 없음 
   	//String[] : 문자열 배열
   	//args : 문자열 배열의 이름
   	//( ) : 매개변수를 입력지점
   	//main : 예약어(KeyWord) , 프로그램의 시작점
   	// { ... } : 함수의 코드 영역
   	public static void main(String[] args) {
   		//실행코드가 들어가는 부분
   		
   		//변수(Variable) : 데이터 저장 메모리 공간 , 램의 특정 공간을 의미(4바이트,8바이트...)
   		
   		//정수형 변수을 선언하자.(실행부에서도 선언,실행 모두 할 수 있다.)
   		// 정수 : 깨끗한 수, 딱 떨어지는 수, 자연수 ex) -2,-1,0,1,2,3, ...
   		// 실수 : 깨끗하지 않은 수 , 딱 떨어지지 않는 수 ex) 1.1 , 2.3 , -3.5
           // int : 정수형 변수선언
   		// i : 변수 이름
   		// = : 대입연산자(오른쪽 값을 왼쪽에 덮어씀)
   		// 10 : 데이터(리터럴)
   		// ; : 실행문 끝에 종료 문구.
   		// 4바이트 메모리공간을 램에 할당하여 10이라는 값을 넣어줌
   		int i = 10;
   		
   		
   		//i라는 변수에 값을 출력
   		System.out.println(i);
   		
   		// 여기서 i는 산술연산자가 아니라 문자열 연결 연산자
   		// i가 문자열로 바뀜
   		System.out.println("i 값은 "+i);
   		
   		
   	}
   
   }
   /*
   10
   i 값은 10
   */
   ```

### 변수와 타입

```java

public class ex2_03 {

	public static void main(String[] args) {
		// 변수 : 데이터를 담는 메모리 공간
		// 정수형 변수 : 정수형 데이터를 담음
		int run = 10;
		
		/*	변수의 이름 짓는 법
		 1. 영소문자로 시작해야 됨
		 2. 숫자나 특수기호로 시작하면 안됨
		 3. 중간에 공백이나 특수기호(%^&*) 안됨
		 4. 자바 예약어(class,int,float)는 쓸 수 없음
		 5. 중간에 _(언더바) 나 대문자를 사용할 수 있음
		 6.	Camel Text 기법으로 변수이름을 짓는다. ex) myCar , nameOfStduents
		 7. 단어를 생략하지 않고 가독성을 위해 풀스펠링으로 짓는다 ex) mC , nOS 로 짓지 않는다.
		*/
		
		
		
		//연습문제
		
		//정수형 변수 count를 선언하고 , 5라는 값을 대입한 후에 count = 5라고 출력하시오
		int count = 5; 
		System.out.println("count = "+count);
		
		
		//정수형 변수 year를 선언하고, 2020값을 대입하고, 변수에 +(산술연산자) 10을 한 후에 "2020의 10년후는 2030"이라고 출력하시오
		int year = 2020;
		System.out.println("2020의 10년후는 "+(year+10));
		
		//상수(Constant) : 변하지 않는 수 ,  변수이면서 값이 한번만 대입되는 변수. final키워드를 이용 , 기본적으로 변수명은 대문자를 사용하고 중간에 _ 사용
		final int MAX_COUNT = 100;
		// MAX_COUNT = 10; 컴파일 에러 발생
		
	}

}
```
